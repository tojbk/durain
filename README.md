git clone local

Create Heroku

heroku create -a durain

initialize git

git add .

push

heroku git push heroku main

heroku config:set APIKEY=You durain API AUTHORIZED_USERS=You TG id BOT_TOKEN=You TG bot Token

start

heroku ps:scale worker=1
